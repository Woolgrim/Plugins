js

$(() => {

    $('.gotop').on( 'click', () => {
        $("html, body").animate({
        scrollTop:0
        }, 0);
    })
})
$(window).on('scroll', () => {

    if ($(this).scrollTop()>500) {
        
        $('.gotop').fadeIn();
    }

    else {
        $('.gotop').fadeOut();
    }
});
